# Git Push Commands for SellApp
# Run these commands in your terminal from the project root (c:\xampp\htdocs\sellapp)

# Step 1: Check status
git status

# Step 2: Add all changed files
git add .

# Step 3: Commit with a descriptive message
git commit -m "Fix asset paths in dashboard views and add notification foreign key fix scripts

- Fixed hardcoded asset paths in admin, manager, salesperson, and technician dashboards
- Added fix_notifications_foreign_keys_simple.sql for cPanel database upload issues
- Added fix_notifications_foreign_keys.php for programmatic fix
- Updated schema.sql to include notifications table definition"

# Step 4: Push to remote repository
# If you haven't set up the remote yet, use:
# git remote add origin ssh://manuelc8@manuelcode.info:7522/home3/manuelc8/sellapp.store

# Then push:
git push -u origin master

# git push -u origin <your-branch-name>
